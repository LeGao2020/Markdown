```c++
#include <iostream>
using namespace std;
int main() {
	double e{ 1 };
	int n = 100000;
	double i;
	int j;
	for (i = 1, j = 1; j <= n; j++, i = i / j) {
		e += i;
		if (j % 10000 == 0) {
			cout << e << endl;
		}
	}
	return 0;
}
```
